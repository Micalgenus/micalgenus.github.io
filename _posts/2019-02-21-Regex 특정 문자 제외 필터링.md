특정 문자를 제외한 문자를 제거하고 싶을때 ^를 사용하면 된다.

{% highlight javascript %}
"basdekrjalskjdblkaweih".replace(/[^abc]/g, ''); // "baaba"

const filter = "abc";
const reg = new RegExp(`[^${filter}]`, 'g');
"basdekrjalskjdblkaweih".replace(reg, ''); // "baaba"
{% endhighlight %}
